Question 1: Speed of Light Calculation
Write a C program that will take 2 floats (Energy and mass) as input and print the Speed of Light as output to 3 decimal places. Einstein’s equation for the theory of relativity is as follows:

E = mc2, where E = energy, m = mass, c = Speed of light.
Sample Input Sample Output
134.5 150.2 0.946
84.9 12.6 2.596
Solution 

#include <stdio.h>
#include <math.h>

int main() {
    float E, m, c;

    scanf("%f %f", &E, &m);

    c = sqrt(E / m);

    printf("%.3f\n", c);

    return 0;
}

Question 2: Rectangle Area and Perimeter Calculation
Write a C program that can calculate the area and perimeter of a rectangle. The system first takes input of a character that can be ‘A’ or ‘P’. If A is entered, the program will compute the area, and if P is entered, the program will compute the perimeter.

Formulas:
  • Area of a rectangle: \( \text{length} \times \text{width} \)
  • Perimeter of a rectangle: \( 2 \times (\text{length} + \text{width}) \)
Sample Input Sample Output
A 5.0 4.0 The area of a rectangle is: 20.000000
P 3.0 2.0 The perimeter of a rectangle is: 10.000000
Solution 

#include <stdio.h>

int main() {
    char choice;
    float length, width;

    scanf("%c %f %f", &choice, &length, &width);

    if (choice == 'A') {
        printf("The area of a rectangle is: %.6f\n", length * width);
    } else if (choice == 'P') {
        printf("The perimeter of a rectangle is: %.6f\n", 2 * (length + width));
    }

    return 0;
}

Question 3: Maximum Value Divisibility Check
Take three integers as input and find the maximum value. If the maximum number is divisible by 2, print “Red Number”, or if it is divisible by 3, print “Blue Number”, or if divisible by both 2 and 3, print “Purple Number”, or if it is divisible by neither, print “White Number”.
Sample Input Sample Output
34 45 40 Blue Number
10 9 7 Red Number
Solution 

#include <stdio.h>

int main() {
    int a, b, c;
    scanf("%d %d %d", &a, &b, &c);

    int maxVal = a;
    if (b > maxVal) maxVal = b;
    if (c > maxVal) maxVal = c;

    if (maxVal % 2 == 0 && maxVal % 3 == 0) {
        printf("Purple Number\n");
    } else if (maxVal % 2 == 0) {
        printf("Red Number\n");
    } else if (maxVal % 3 == 0) {
        printf("Blue Number\n");
    } else {
        printf("White Number\n");
    }

    return 0;
}

Question 4: Maximum Value with Addition and Multiplication
Write a C program that will take three integer numbers as input, and calculate the maximum value after using exactly one addition and exactly one multiplication operation among those numbers.
Hints: Compute values for all three possible combinations \(a + b \times c\), \(b + a \times c\), and \(c + a \times b\) and find the maximum value.
Sample Input Sample Output
1 4 7 Maximum value: 29
-5 0 3 Maximum value: 3
-3 -2 -9 Maximum value: 25
Solution 

#include <stdio.h>

int main() {
    int a, b, c;
    scanf("%d %d %d", &a, &b, &c);

    int val1 = a + b * c;
    int val2 = b + a * c;
    int val3 = c + a * b;

    int maxVal = val1;
    if (val2 > maxVal) maxVal = val2;
    if (val3 > maxVal) maxVal = val3;

    printf("Maximum value: %d\n", maxVal);

    return 0;
}

Question 5: Print Digits from Last Digit to 9
Write a program that will take a positive integer as input, find the last digit, and print all the digits from the last digit to digit 9. You must use switch-case statements and the last digit as its input.
Sample Input Sample Output
54 456789
90 0123456789
9 9
16 6789
Solution 

#include <stdio.h>

int main() {
    int num;
    scanf("%d", &num);
    
    int lastDigit = num % 10;

    switch (lastDigit) {
        case 0: printf("0123456789\n"); break;
        case 1: printf("123456789\n"); break;
        case 2: printf("23456789\n"); break;
        case 3: printf("3456789\n"); break;
        case 4: printf("456789\n"); break;
        case 5: printf("56789\n"); break;
        case 6: printf("6789\n"); break;
        case 7: printf("789\n"); break;
        case 8: printf("89\n"); break;
        case 9: printf("9\n"); break;
        default: printf("Invalid input\n"); break;
    }

    return 0;
}

Question 1: Volume and Surface Area of a Sphere
The volume of a sphere is given by the formula \(V = \frac{4}{3} \pi r^3\), and the surface area of a sphere is given by the formula \(A = 4 \pi r^2\), where \(r\) is the radius of the sphere. Write a program that will take the radius of a sphere as input and compute and print the volume and surface area of the sphere. (\(\pi = 3.1416\))
Sample Input Sample Output
10.5 Volume = 4849.06, Area = 1385.45
12.9 Volume = 8992.05, Area = 2091.17
Solution 

#include <stdio.h>
#include <math.h>

int main() {
    float radius, volume, area;
    const float pi = 3.1416;

    scanf("%f", &radius);

    volume = (4.0/3.0) * pi * pow(radius, 3);
    area = 4 * pi * pow(radius, 2);

    printf("Volume = %.2f, Area = %.2f\n", volume, area);

    return 0;
}

Question 2: Evaluate Function \(f(x, y)\)
A function \(f(x, y)\) is defined as follows: \[ f(x, y) = \begin{cases} x^3 + 5xy, & \text{if } x < 0 \text{ and } y < 0 \\ 4y, & \text{if } x < 0 \text{ and } y> 0 \\ \frac{1}{x + y}, & \text{if } x \geq 0 \text{ and } y \geq 0 \\ \end{cases} \] Write a C program to evaluate \(f(x, y)\) following the above definition. For values not in the mentioned range, your program should output “Undefined”.
Sample Input Sample Output
-3.8 -2.2 -13.072
-0.6 0 Undefined
5 2 0.143
Solution 

#include <stdio.h>

float f(float x, float y) {
    if (x < 0 && y < 0) {
        return x * x * x + 5 * x * y;
    } else if (x < 0 && y > 0) {
        return 4 * y;
    } else if (x >= 0 && y >= 0) {
        return 1.0 / (x + y);
    } else {
        return -1; // Undefined for all other cases
    }
}

int main() {
    float x, y;
    scanf("%f %f", &x, &y);

    float result = f(x, y);
    if (result == -1) {
        printf("Undefined\n");
    } else {
        printf("%.3f\n", result);
    }

    return 0;
}

        min = c;
    }

Write a C program that asks the user to input three numbers representing the lengths of the sides of a triangle. Using if/else statements, determine and print whether the triangle is valid or not. If the triangle is valid, print “Valid Triangle.” If the triangle is invalid, print “Invalid Triangle.”
Hints: A triangle is valid if the sum of any two sides is greater than the third side.
Sample Input Sample Output
2 9 10 Valid Triangle.
1 2 3 Invalid Triangle.
Solution 

#include <stdio.h>

int main() {
    int a, b, c;
    scanf("%d %d %d", &a, &b, &c);

    if (a + b > c && a + c > b && b + c > a) {
        printf("Valid Triangle.\n");
    } else {
        printf("Invalid Triangle.\n");
    }

    return 0;
}

Question 5: Student ID Operation
Write a program that will take the last 4 digits of your student ID and an operator as input. The program will determine the last digit of your student ID and perform an operation on that digit three times using switch case statements.
Sample Input Sample Output
1145 * 5 * 5 * 5 = 125
1123 + 3 + 3 + 3 = 9
1128 - 8 - 8 - 8 = -8
1122 ? The input is invalid
Solution 

#include <stdio.h>

int main() {
    int studentID, lastDigit;
    char operator;

    scanf("%d %c", &studentID, &operator);

    lastDigit = studentID % 10;

    switch(operator) {
        case '+':
            printf("%d + %d + %d = %d\n", lastDigit, lastDigit, lastDigit, lastDigit + lastDigit + lastDigit);
            break;
        case '-':
            printf("%d - %d - %d = %d\n", lastDigit, lastDigit, lastDigit, lastDigit - lastDigit - lastDigit);
            break;
        case '*':
            printf("%d * %d * %d = %d\n", lastDigit, lastDigit, lastDigit, lastDigit * lastDigit * lastDigit);
            break;
        default:
            printf("The input is invalid\n");
    }

    return 0;
}
